Key CSS patterns for integration
/* CHAMFER CLIP-PATH FORMULA */
.chamfer-md {
  --chamfer: 10px;
  clip-path: polygon(
    var(--chamfer) 0,              /* top-left chamfer */
    calc(100% - var(--chamfer)) 0, /* top-right chamfer start */
    100% var(--chamfer),           /* top-right chamfer end */
    100% calc(100% - var(--chamfer)), /* bottom-right chamfer start */
    calc(100% - var(--chamfer)) 100%, /* bottom-right chamfer end */
    var(--chamfer) 100%,           /* bottom-left chamfer start */
    0 calc(100% - var(--chamfer)), /* bottom-left chamfer end */
    0 var(--chamfer)               /* back to top-left */
  );
}

/* REGISTRATION BRACKET */
.bracket::before { width: 12px; height: 2px; }
.bracket::after { width: 2px; height: 12px; }

/* HEXAGONAL SHAPE */
.hex {
  clip-path: polygon(25% 0%, 75% 0%, 100% 50%, 75% 100%, 25% 100%, 0% 50%);
}
